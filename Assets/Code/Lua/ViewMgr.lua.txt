--注册UI控制器 （动态添加）------------------


local ViewMgr = {}
local viewList = {}

function ViewMgr:LoadView(viewName, model, func)
    CS.LuaHelperManager.Instance:LoadUI(viewName, model, func);
    viewList[viewName] = model
    return ViewMgr
end

function ViewMgr:GetView(viewName)
    return viewList[viewName]
end

return ViewMgr
