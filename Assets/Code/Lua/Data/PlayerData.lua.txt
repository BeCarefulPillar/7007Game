M = {}

function M:SetName(name)
    self.name = name
end

local function toClass(t)
    t = t or {}
    return M.makeObject({
        id = t.id,
        name = t.name or "",
        hero = t.hero or {}
    })
end

function M.new(playerId)
    local player = toClass()
    player.id = playerId
    player.name = "xxx"
    table.insert(player.hero,{name = "zz", hp = 1000, atk = 1000})
    table.insert(player.hero,{name = "bb", hp = 8888, atk = 500})
    return player
end

function M.newByTable(t)
    local player = toClass(t)
    return player
end

function M.makeObject(t)
    return setmetatable(t, {__index=M})
end

return M
