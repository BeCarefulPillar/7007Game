-- CtrlMgr.lua 脚本 初始化所有的控制器脚本、存储到table表中
require "Define"

--注册UI控制器 （动态添加）------------------
require "Modules/UIRoot/UIRootCtrl"

CtrlMgr = {}
local this = CtrlMgr
local ctrlList = {}

function CtrlMgr.Init()
    ctrlList[CtrlNames.UIRootCtrl] = UIRootCtrl.New()
    return this
end

function CtrlMgr.GetCtrl(ctrlName)
    return ctrlList[ctrlName]
end
