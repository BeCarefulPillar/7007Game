M = {}

local function toClass(t)
    t = t or {}
    return M.makeObject({
        typeId = t.typeId,
        name = t.name or "",
        hp = t.hp or 1000,
        atk = t.atk or 10,
    })
end

function M.new()
    local physic = toClass()
    return physic
end

function M.newByTable(t)
    local physic = toClass(t)
    return physic
end

function M.makeObject(t)
    return setmetatable(t, {__index=M})
end

return M
