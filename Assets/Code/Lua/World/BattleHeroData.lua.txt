
local M = {}

function M:update()


    print("111111111111111")


end

local function toClass(t)
    t = t or {}
    return M.makeObject({
        typeId = t.typeId,
        name = t.name or "",
        hp = t.hp or 1000,
        atk = t.atk or 10,
        size = t.size or {5,5}
    })
end

function M.new(hero)
    local hero = toClass(hero)
    return hero
end

function M.makeObject(t)
    return setmetatable(t, {__index=M})
end

return M
