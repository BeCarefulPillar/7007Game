local function toClass(t)
    t = t or {}
    return M.makeObject({
        size = t.size or {6,5},  --目前是meshrender大小 之后这个里面的数据全部写到表中
        pos = t.pos or {8,-5},
    })
end

function M.new()
    return toClass()
end

function M.makeObject(t)
    return setmetatable(t, {__index=M})
end

return M
